//Unsigned Right Shift

import java.io.*;
public class Main {
	public static void main (String[] args) {
	    
	    int x = -2;
	    
	    System.out.println(x >>> 1);
	    
	}
}

//Right Shift with Negative Number

import java.io.*;
public class Main {
	public static void main (String[] args) {
	    
	    int x = -2;
	    
	    System.out.println(x >> 1);
	    
	}
}

//Right Shift

import java.io.*;
public class Main {
	public static void main (String[] args) {
	    
	    int x = 33;
	    
	    System.out.println(x >> 2);
	    
	}
}
