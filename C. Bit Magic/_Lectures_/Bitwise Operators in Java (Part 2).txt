//Left Shift with Negative Number
import java.io.*;

public class Main {
	public static void main (String[] args) {
	    
	    int x = -1;
	    
	    System.out.println(x << 1);
	    
	}
}

//Left Shift
import java.io.*;

public class Main {
	public static void main (String[] args) {
	    
	    int x = 3;
	    
	    System.out.println(x << 1);
	    
	    System.out.println(x << 2);
	    
	    int y = 4;
	    
	    int z = (x << y);
	    
	    System.out.println(z);
	}
}

//Bitwise Not
import java.io.*;

public class Main {
	public static void main (String[] args) {
	    
	    int x = 1;
	    
		System.out.println(~ x);
	}
}
